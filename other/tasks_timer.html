void write_ds1307(unsigned short address, unsigned short w_data){  //---- WRITE
 I2C1_Start();           // issue I2C start signal
 I2C1_Wr(0xD0);          // Device Address + W = 0xD0
 I2C1_Wr(address);       // send byte (address of DS1307 location)
 I2C1_Wr(w_data);        // send data (data to be written)
 I2C1_Stop();            // issue I2C stop signal
}

unsigned short read_ds1307(unsigned short address){                //---- READ
 I2C1_Start();
 I2C1_Wr(0xD0);          // Device Address + W = 0xD0
 I2C1_Wr(address);
 I2C1_Repeated_Start();
 I2C1_Wr(0xD1);          // Device Address + R = 0xD1
 readData=I2C1_Rd(0);    // Read Data from DS1307
 I2C1_Stop();
 return(readData);
}

unsigned char BCD2UpperCh(unsigned char bcd){                      //---- BCD
 return ((bcd >> 4) + '0');
}

unsigned char BCD2LowerCh(unsigned char bcd){                      //---- BCD
 return ((bcd & 0x0F) + '0');
}

void dataSave(){           // ------------ Data saving procedure -------------
     for(i=0;i<10;i++){
         write_ds1307(8+i,state[i]);                     // Saving item's state
         write_ds1307(18+i,secCount[i]>>8);              // Saving high second
         write_ds1307(28+i,secCount[i]&255);             // Saving low second
     }
     write_ds1307(42,beepCount);                         // Saving Beep counter
}                          // ------------------------------------------------

void powerCut(){           // --------- Saving data when black-out -----------
     Lcd_Out(1,1,CopyConst2Ram(msg,powerError));
     Lcd_Out(2,1,CopyConst2Ram(msg,blank16));
     dataSave();                                         // Data save function
     write_ds1307(38,read_ds1307(1));                    // Saving Minutes
     write_ds1307(40,read_ds1307(2));                    // Saving Hours
     Lcd_Out(2,1,CopyConst2Ram(msg,allDataSaved));
     Sound_Play(5000,60);                                // Sound alarm
     Delay_ms(10);
     Sound_Play(4000,60);
     Delay_ms(60);
     Sound_Play(5000,60);
     Delay_ms(10);
     Sound_Play(4000,60);
     Delay_ms(60);
     Sound_Play(5000,60);
     Delay_ms(10);
     Sound_Play(4000,60);
     while(PORTB.f5);                                    // While low power
     Delay_ms(1000);
     Lcd_Cmd(_LCD_CLEAR);
}                          // ------------------------------------------------

void last_off(){           // --------- Show time of last Power Off ----------
     Lcd_Out(1,1,CopyConst2Ram(msg,lastPowerOff));
     Lcd_Out(2,1,CopyConst2Ram(msg,blank16));
     msg[0] =  BCD2UpperCh(read_ds1307(40));             // Hour
     msg[1] =  BCD2LowerCh(read_ds1307(40));             // Hour
     msg[2] = ':';
     msg[3] =  BCD2UpperCh(read_ds1307(38));             // Minute
     msg[4] =  BCD2LowerCh(read_ds1307(38));             // Minute
     msg[5] = '\0';
     Lcd_Out(2,6,msg);
     Delay_ms(delayLong);
     Lcd_Cmd(_LCD_CLEAR);
}                          // ------------------------------------------------

void interrupt(){          // ---------- Current task's second ++ ------------
     if(INTCON.INTF){
         INTCON.INTF = 0;
         porta.f0 = ~porta.f0;
         tempCount++;
         for(i=0; i<10; i++){
           if(state[i] == 1){
             secCount[i] ++;
           }
         }
     }
}                          // ------------------------------------------------

void main() {              // ---------------- Main function -----------------
     PORTA = 0x00;
     PORTB = 0x00;
     PORTC = 0x00;                         // Reseting ports
     PORTD = 0x00;
     PORTE = 0x00;

     TRISA = 0b00100000;
     TRISB = 255;
     TRISC = 0;                            // Setting direction of ports
     TRISD = 0;
     TRISE = 255;

     ANSEL  = 0;                           // Configure all pins as digital I/O
     ANSELH = 0;
     C1ON_bit = 0;                         // Disable comparators
     C2ON_bit = 0;

     OPTION_REG = 0b01111111;              // Enable pull-up resistors in portb
     WPUB = 0b00011111;                    // Connects pull-ups

     I2C1_Init(100000);                    // Initializing I2C at 100KHz
     Lcd_Init();                           // Initializing LCD
     Sound_Init(&PORTD,7);                 // Initializing sound module
  
     Lcd_Cmd(_LCD_CLEAR);                  // Clear display
     Lcd_Cmd(_LCD_CURSOR_OFF);             // Cursor off
     PORTA.F0 = 1;                         // 7Segment DOT
     PORTA.F4 = 1;                         // LCD Backligh ON
     PORTD = 191;                          // 7Segment print "-"
     beepCount = 0;                        // Beep flag counter
     item = 10;
     Sound_Play(1500,200);                 // Beep @ start
     
         // -- Initialising DS1307 (Possible ONLY when the programme starts) --
     if(PORTB.f4 == 0) {
           write_ds1307(0,0x80);  // Reset second to 0 and stop Oscillator (0)
           write_ds1307(1,0x00);  // write minutes
           write_ds1307(2,0x12);  // write hours
           write_ds1307(3,0x01);  // write day of week 1:Sunday
           write_ds1307(4,0x01);  // write date
           write_ds1307(5,0x03);  // write month
           write_ds1307(6,0x14);  // write year : 20XX
           write_ds1307(7,0b00010000); // SQWE output at 1 Hz
           write_ds1307(0,0x00);  // Reset second to 0 and start Oscillator (1)
     }   // ------------------------------------------------

     Lcd_Out(1,3,CopyConst2Ram(msg,tasksTimer));
     Delay_ms(delayShort);
     Lcd_Cmd(_LCD_CLEAR);

     for (i=0; i<10; i++){              // Reseting variables
         secCount[i] = 0;
         state[i] = 0;
     }
     
     INTCON = 0b10010000;               // Enable RB0/INT interrupt
     
     last_off();                        // Show time of last Power Off
     load_all();                        // Saved data load function
//     calcTime();                        // Calculate running time during Off

     while(1){
              read_time();              // Read time from DS1307
              
              if((msg[15]%2==0)&&(incData==0)){                  // Saving data
                    dataSave();                                  // @ even mins
              }

              if(msg[11]==49){                                   // Beep
                    if (msg[12]==48&&msg[14]==48&&msg[15]==48&&beepCount.f0==0){
                       Sound_Play(5000,200);
                       Sound_Play(0,150);
                       Sound_Play(5000,200);                     
                       beepCount.f0 = 1;                         // Flag @10:00
                    }
                    if (msg[12]==51&&msg[14]==48&&msg[15]==48&&beepCount.f1==0){
                       Sound_Play(5000,200);
                       Sound_Play(0,150);
                       Sound_Play(5000,200);                     
                       beepCount.f1 = 1;                         // Flag @13:00
                    }
                    if (msg[12]==51&&msg[14]==51&&msg[15]==48&&beepCount.f2==0){
                       Sound_Play(5000,200);
                       Sound_Play(0,150);
                       Sound_Play(5000,200);
                       beepCount.f2 = 1;                         // Flag @13:30
                    }
                    if (msg[12]==53&&msg[14]==51&&msg[15]==48&&beepCount.f3==0){
                       Sound_Play(5000,200);
                       Sound_Play(0,150);
                       Sound_Play(5000,200);
                       beepCount.f3 = 1;                         // Flag @15:30
                    }
              } else
                    beepCount = 0;

              if (Button(&PORTB,1,deb_time,0)){                  // Item ++
                    item++;
                    if (item == 11){
                        Lcd_Cmd(_LCD_CLEAR);
                        item = 0;
                    }
              }
              if (Button(&PORTB,2,deb_time,0) && (item != 10)){  // Start / Stop
                    if(state[item] == 0){
                        state[item] = 1;
                    } else if (state[item] == 1){
                        state[item] = 0;
                    }
              }
              if (Button(&PORTB,2,deb_time,0) && (item == 10)){  // All Reset
                    reset_all();
              }
              if (Button(&PORTB,3,deb_time,0) && (item == 10)){  // Last Off
                    last_off();
              }
              if (Button(&PORTB,3,deb_time,0) && (item != 10)){  // Item Reset
                    reset_one();
              }
              if (Button(&PORTB,4,deb_time,0) && (item == 10)){  // Setting Menu
                    set_menu();
              }
              if (Button(&PORTB,4,deb_time,0) && (item != 10)){  // Toggle act.
                    if (PORTA.f1){
                        item++;
                        while((item > 9)||(state[item] == 0)){
                            item++;
                        }
                    }
              }
              
              segment();                // 7Segment display
              print_state();            // "Stopped" or "Running"
              calc_minutes();           // Calculate current item's minutes

              if (PORTB.f5 == 1){       // Power supply check
                 PORTA.f4 = 0;          // LCD Backlight OFF
                 powerCut();
                 PORTA.f4 = 1;          // LCD Backlight ON
              }

              if ((state[item] == 0)||(item == 10)){             // Green LED
                 PORTA.f2 = 0;
              } else {
                 PORTA.f2 = 1;
              }

              } else {
                               PORTA.f1 = 0;
                               PORTA.f3 = 1;
              }
     }
